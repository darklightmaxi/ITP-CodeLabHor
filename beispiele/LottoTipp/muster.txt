public static boolean contains(int[] a, int search) {
    for (int num : a) {
        if (num == search) {
            return true;
        }
    }
    return false;
}

public static int[] getLottoTipp(){

    int[] a = new int[6];

    for (int i = 0; i < 6; i++) {
        Random random = new Random();
        int randomNumber = random.nextInt(45) + 1;
        if (!contains(a, randomNumber)){
            a[i] = randomNumber;
        }
    }

    return a;
}
